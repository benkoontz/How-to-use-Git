// git commands
git init // initialize local git repository
git add <file> // add file(s) to index
git rm <file>// remove a file
git status // check status of working tree
git commit // commit changes in index
git touch <file> // add a file
git push // push to remote repository such as github
git pull // pull latest version from remote repository
git clone // clone repository into a new directory
git branch <file> // create new branch
git checkout <file> // move to that branch
git merge // used to merge files
